interface定义函数

interface User {
    name: string
    age: number
}
// 定义单个函数，用type定义是个不错的选择
type UserFunc = (user: User) => string
const getUserInfo: UserFunc = (user) => {
    return `name: ${user.name}，age: ${user.age}`
}
getUserInfo({ name: 'ff', age: 23 })

typeof 探查某个dom/关键字的类型

// 想知道某个dom，关键字(setTimeout)的类型是什么，可用typeof
let div = document.createElement('div');
// HTMLDivElement
type divType = typeof div   

let timer = setTimeout(() => console.log('st'), 2)
// NodeJS.Timeout
type timerType = typeof timer

ts高级类型

// 剔除某个键 Omit
let person = {
    name: 'name',
    hoppy: ['eat', 'sleep']
}
type user = Omit<typeof person, 'name'>
