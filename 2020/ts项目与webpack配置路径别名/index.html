webpack与ts项目配置路径别名的问题

1、can not find xxx module

解决方案：

// 1、首先安装 awesome-typescript-loader  
cnpm i awesome-typescript-loader -S  
// 2、webpack.config.js
const { TsConfigPathsPlugin } = require('awesome-typescript-loader');
resolve: {
    plugins: [
        new TsConfigPathsPlugin()
    ]
}
// 3、tsconfig.json
{
  "compilerOptions": {
        ...
        "rootDir": ".",
        "paths": {
            "@src/*": [
                "src/lib/*"
            ],
            "@util/*": [
                "src/util/*"
            ]
        }
    }
}
